Источником данных может выступать либо массив данных (dataSource), либо url, с которого данные загружаются (dataUrl). Если установлен dataUrl, то isLoadDataByPostQuery определяет каким образом загружать данные: методом post как PostQuery/PostSearch или get-запросом. По-умолчанию, isLoadDataByPostQuery = true.

Properties

Name Description
@Input()
disabled: boolean		 Доступность для редактирования: true - включить режим readonly
@Input()
placeholder: string		 Placeholder
@Input()
value: any		 Выбранное значение
@Input()
valueField: string		 Свойство в элементе источника данных, которое будет использоваться как выбранное значение value
@Input()
displayField: string = 'name'		 Свойство в элементе источника данных, которое будет использоваться для отображения
@Input()
dataSource: any[] = []		 Массив, который используется как источник данных
@Input()
dataUrl: string		 Url, который используется как источник данных
@Input()
isLoadDataByPostQuery = true		 Как загружать данные с dataUrl. Если true, то данные загружаются методом post как к PostQuery/PostSearch. Если false - выполняется get запрос
@Input()
dataFilter: any		 Фильтр, передаваемый в dataUrl, когда isLoadDataByPostQuery = true
@Output()
filterField: string = 'nameContains'		 По какому полю, вводимый пользователем текст, фильтрует данные
@Output()
valueChange		 Событие, вызываемое при выборе значения
